% --- Coordinateur ---
\subsubsection{Module Coordination}
\begin{figure}[H]
    \centering
    \pumlsvg[\textwidth]{uc_coordination}
    \caption{Cas d'utilisation : Coordination des soutenances}
\end{figure}

\begin{table}[H]
    \centering
    \small
    \begin{tabular}{lp{10cm}}
        \toprule
        \rowcolor{primary!10}
        \textbf{Cas d'utilisation} & \textbf{Description}                                                                                       \\
        \midrule
        Assigner jurys             & Constitution des commissions (président, rapporteur, examinateur) avec recherche d'équilibre de la charge. \\
        \addlinespace
        Planifier soutenances      & Création des sessions, définition des dates, horaires et affectation des salles aux étudiants.             \\
        \addlinespace
        Valider planning           & Confirmation finale du planning après traitement des alertes et détection des conflits.                    \\
        \addlinespace
        Gérer étudiants            & Administration des dossiers étudiants (inscription, filière, niveau académique).                           \\
        \addlinespace
        Gérer les projets          & Gestion des thèmes, titres de PFE et association entre étudiants, projets et encadrants.                   \\
        \addlinespace
        Importer listes            & Chargement massif de données (CSV/Excel) pour les étudiants, enseignants et projets.                       \\
        \addlinespace
        Consulter planning         & Visualisation globale filtrable (par date, salle, enseignant, filière ou étudiant).                        \\
        \addlinespace
        Rechercher / Filtrer       & Recherche multicritère avancée pour la supervision et le suivi des dossiers.                               \\
        \addlinespace
        Générer convocations       & Production automatique des documents officiels (PDF) pour les étudiants et les membres de jury.            \\
        \addlinespace
        Générer planning général   & Production du planning complet et de la liste de présence pour la session de soutenance.                   \\
        \addlinespace
        Consulter rapports         & Accès aux synthèses, statistiques d'activité et procès-verbaux (PV) simplifiés.                            \\
        \addlinespace
        Vérifier conflits          & Détection des chevauchements (salles, enseignants), doubles planifications et doublons.                    \\
        \bottomrule
    \end{tabular}
    \caption{Description des cas d'utilisation -- Coordinateur}
\end{table}
